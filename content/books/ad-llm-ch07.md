#### 数据去重的三层策略

```python
# 自动驾驶数据三层去重 pipeline
class DrivingDataDedup:
    def __init__(self):
        self.frame_dedup = FrameLevelDedup(
            model="dinov2-large",
            threshold=0.15  # 余弦距离阈值
        )
        self.segment_dedup = SegmentLevelDedup(
            dtw_threshold=2.0,  # 轨迹DTW距离
            scene_clusters=50   # 语义场景聚类数
        )
        self.scene_dedup = SceneLevelDedup(
            modalities=["vision", "lidar", "trajectory", "map"],
            difficulty_weight=True  # 保留高困难度样本
        )

    def pipeline(self, raw_data):
        # 帧级: ~50% 压缩
        stage1 = self.frame_dedup.process(raw_data)
        # 片段级: ~30% 压缩
        stage2 = self.segment_dedup.process(stage1)
        # 场景级: ~20% 压缩
        final = self.scene_dedup.process(stage2)
        # 总压缩率: ~70%, 保留 30% 高价值数据
        return final
```
